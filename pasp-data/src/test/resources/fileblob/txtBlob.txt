<?xml version="1.0" encoding="UTF-8"?>
<configuration scan="true">
	<!-- 应用属性 -->
	<property resource="conf/pasp.properties" />
	<!-- 日志输出目录 -->
	<property name="home" value="${user.dir}/logs"/>
	<!-- 日志打印格式 -->
	<property name="pattern" value="%d{yyyyMMddHHmmss:SSS}!${app.name}:${module.name} [%thread]#%logger{36}:%-5level>%msg%n"/>
	
	<appender name="STDOUT" class="ch.qos.logback.core.ConsoleAppender">
		<encoder>
			<pattern>${pattern}</pattern>
		</encoder>
	</appender>

	<appender name="INFO"
		class="ch.qos.logback.core.rolling.RollingFileAppender">
		<File>${home}/${app.name}.${module.name}.info.log</File>
		<rollingPolicy class="ch.qos.logback.core.rolling.TimeBasedRollingPolicy">
			<FileNamePattern>
				${home}/${app.name}.${module.name}.%d{yyyyMMdd}.info.log.zip
			</FileNamePattern>
			<maxHistory>90</maxHistory>
			<cleanHistoryOnStart>true</cleanHistoryOnStart>
		</rollingPolicy>
		<encoder>
			<pattern>${pattern}</pattern>
		</encoder>
		<filter class="ch.qos.logback.classic.filter.LevelFilter">
			<level>INFO</level>
			<onMatch>ACCEPT</onMatch>
			<onMismatch>NEUTRAL</onMismatch>
		</filter>
		<filter class="ch.qos.logback.classic.filter.LevelFilter">
			<level>WARN</level>
			<onMatch>ACCEPT</onMatch>
			<onMismatch>DENY</onMismatch>
		</filter>
	</appender>

	<appender name="ERROR"
		class="ch.qos.logback.core.rolling.RollingFileAppender">
		<File>${home}/${app.name}.${module.name}.error.log</File>
		<rollingPolicy class="ch.qos.logback.core.rolling.TimeBasedRollingPolicy">
			<FileNamePattern>
				${home}/${app.name}.${module.name}.%d{yyyyMMdd}.error.log.zip
			</FileNamePattern>
			<maxHistory>90</maxHistory>
			<cleanHistoryOnStart>true</cleanHistoryOnStart>
		</rollingPolicy>
		<encoder>
			<pattern>${pattern}</pattern>
		</encoder>
		<filter class="ch.qos.logback.classic.filter.LevelFilter">
			<level>ERROR</level>
			<onMatch>ACCEPT</onMatch>
			<onMismatch>DENY</onMismatch>
		</filter>
	</appender>

	<appender name="TRACE"
		class="ch.qos.logback.core.rolling.RollingFileAppender">
		<File>${home}/${app.name}.${module.name}.trace.log</File>
		<rollingPolicy class="ch.qos.logback.core.rolling.TimeBasedRollingPolicy">
			<FileNamePattern>
				${home}/${app.name}.${module.name}.%d{yyyyMMdd}.trace.log.zip
			</FileNamePattern>
			<maxHistory>90</maxHistory>
			<cleanHistoryOnStart>true</cleanHistoryOnStart>
		</rollingPolicy>
		<encoder>
			<pattern>${pattern}</pattern>
		</encoder>
		<filter class="ch.qos.logback.classic.filter.LevelFilter">
			<level>TRACE</level>
			<onMatch>ACCEPT</onMatch>
			<onMismatch>NEUTRAL</onMismatch>
		</filter>
		<filter class="ch.qos.logback.classic.filter.LevelFilter">
			<level>DEBUG</level>
			<onMatch>ACCEPT</onMatch>
			<onMismatch>DENY</onMismatch>
		</filter>
	</appender>
	
	<root level="TRACE">
		<appender-ref ref="STDOUT" />
		<!-- <appender-ref ref="INFO" />
		<appender-ref ref="ERROR" />
		<appender-ref ref="TRACE" /> -->
	</root>
	
</configuration>
